"use client"

import { useState } from "react"
import * as Tabs from "@radix-ui/react-tabs"
import { ArrowLeft, Copy, Check, Smartphone, Key, Shield, Download, RefreshCw } from "lucide-react"

interface TwoFactorSetupProps {
  onBack?: () => void
  onComplete?: () => void
}

export function TwoFactorSetup({ onBack, onComplete }: TwoFactorSetupProps) {
  const [currentStep, setCurrentStep] = useState(1)
  const [verificationCode, setVerificationCode] = useState("")
  const [isVerifying, setIsVerifying] = useState(false)
  const [isSetupComplete, setIsSetupComplete] = useState(false)
  const [copiedSecret, setCopiedSecret] = useState(false)
  const [activeMethod, setActiveMethod] = useState("qr")

  const secretKey = "JBSWY3DPEHPK3PXPJBSWY3DPEHPK3PXP"
  const backupCodes = [
    "1a2b-3c4d",
    "5e6f-7g8h",
    "9i0j-1k2l",
    "3m4n-5o6p",
    "7q8r-9s0t",
    "1u2v-3w4x",
    "5y6z-1a2b",
    "3c4d-5e6f",
    "7g8h-9i0j",
    "1k2l-3m4n",
  ]

  const copyToClipboard = (text: string) => {
    navigator.clipboard.writeText(text)
    setCopiedSecret(true)
    setTimeout(() => setCopiedSecret(false), 2000)
  }

  const handleVerification = async () => {
    if (verificationCode.length !== 6) return

    setIsVerifying(true)
    // 模擬驗證過程
    await new Promise((resolve) => setTimeout(resolve, 1500))
    setIsVerifying(false)
    setIsSetupComplete(true)
    setCurrentStep(3)
  }

  const downloadBackupCodes = () => {
    const content = `雙重驗證備用恢復碼\n生成時間: ${new Date().toLocaleString()}\n\n${backupCodes.join("\n")}\n\n請將這些備用碼保存在安全的地方。每個備用碼只能使用一次。`
    const blob = new Blob([content], { type: "text/plain" })
    const url = URL.createObjectURL(blob)
    const a = document.createElement("a")
    a.href = url
    a.download = "2fa-backup-codes.txt"
    document.body.appendChild(a)
    a.click()
    document.body.removeChild(a)
    URL.revokeObjectURL(url)
  }

  return (
    <div className="max-w-2xl mx-auto space-y-6">
      {/* Header */}
      <div className="flex items-center gap-4">
        {onBack && (
          <button onClick={onBack} className="p-2 rounded-md hover:bg-gray-100 transition-colors">
            <ArrowLeft className="h-5 w-5" />
          </button>
        )}
        <div>
          <h1 className="text-2xl font-semibold tracking-tight text-gray-900">設定雙重驗證</h1>
          <p className="text-sm text-gray-500 mt-1">為您的帳戶增加額外的安全保護層。</p>
        </div>
      </div>

      {/* Progress Steps */}
      <div className="bg-white border border-gray-200 rounded-lg shadow-sm p-6">
        <div className="flex items-center justify-between mb-6">
          {[1, 2, 3].map((step) => (
            <div key={step} className="flex items-center">
              <div
                className={`w-8 h-8 rounded-full flex items-center justify-center text-sm font-medium ${
                  currentStep >= step ? "bg-black text-white" : "bg-gray-200 text-gray-500"
                }`}
              >
                {currentStep > step ? <Check className="h-4 w-4" /> : step}
              </div>
              {step < 3 && <div className={`w-16 h-0.5 mx-2 ${currentStep > step ? "bg-black" : "bg-gray-200"}`} />}
            </div>
          ))}
        </div>

        <div className="text-center">
          <p className="text-sm text-gray-600">
            {currentStep === 1 && "步驟 1: 選擇設定方式"}
            {currentStep === 2 && "步驟 2: 驗證設定"}
            {currentStep === 3 && "步驟 3: 保存備用碼"}
          </p>
        </div>
      </div>

      {/* Step 1: Setup Method */}
      {currentStep === 1 && (
        <div className="bg-white border border-gray-200 rounded-lg shadow-sm">
          <div className="p-6 border-b border-gray-200">
            <div className="flex items-center gap-2">
              <Shield className="h-5 w-5 text-gray-500" />
              <h2 className="text-lg font-medium text-gray-900">選擇設定方式</h2>
            </div>
            <p className="text-sm text-gray-500 mt-1">選擇您偏好的雙重驗證設定方式。</p>
          </div>

          <div className="p-6">
            <Tabs.Root value={activeMethod} onValueChange={setActiveMethod}>
              <Tabs.List className="grid w-full grid-cols-2 bg-gray-100 rounded-lg p-1">
                <Tabs.Trigger
                  value="qr"
                  className="flex items-center justify-center gap-2 px-3 py-2 text-sm font-medium rounded-md transition-all data-[state=active]:bg-white data-[state=active]:shadow-sm"
                >
                  <Smartphone className="h-4 w-4" />
                  掃描 QR Code
                </Tabs.Trigger>
                <Tabs.Trigger
                  value="manual"
                  className="flex items-center justify-center gap-2 px-3 py-2 text-sm font-medium rounded-md transition-all data-[state=active]:bg-white data-[state=active]:shadow-sm"
                >
                  <Key className="h-4 w-4" />
                  手動輸入金鑰
                </Tabs.Trigger>
              </Tabs.List>

              <div className="mt-6">
                <Tabs.Content value="qr" className="space-y-4">
                  <div className="text-center">
                    <h3 className="text-lg font-medium text-gray-900 mb-2">掃描 QR Code</h3>
                    <p className="text-sm text-gray-500 mb-6">
                      使用您的驗證器應用程式（如 Google Authenticator、Authy 等）掃描下方的 QR Code。
                    </p>

                    {/* QR Code */}
                    <div className="flex justify-center mb-6">
                      <div className="w-48 h-48 bg-white border-2 border-gray-200 rounded-lg flex items-center justify-center shadow-sm">
                        <div className="w-40 h-40 bg-gray-100 border-2 border-dashed border-gray-300 rounded-lg flex items-center justify-center">
                          <div className="text-center">
                            <div className="w-16 h-16 bg-black rounded-lg mx-auto mb-2 flex items-center justify-center">
                              <div className="grid grid-cols-3 gap-1">
                                {[...Array(9)].map((_, i) => (
                                  <div
                                    key={i}
                                    className={`w-1 h-1 ${Math.random() > 0.5 ? "bg-white" : "bg-transparent"}`}
                                  />
                                ))}
                              </div>
                            </div>
                            <p className="text-xs text-gray-500">QR Code</p>
                          </div>
                        </div>
                      </div>
                    </div>

                    <div className="bg-blue-50 border border-blue-200 rounded-lg p-4">
                      <h4 className="text-sm font-medium text-blue-900 mb-2">推薦的驗證器應用程式：</h4>
                      <ul className="text-sm text-blue-800 space-y-1">
                        <li>• Google Authenticator</li>
                        <li>• Microsoft Authenticator</li>
                        <li>• Authy</li>
                        <li>• 1Password</li>
                      </ul>
                    </div>
                  </div>
                </Tabs.Content>

                <Tabs.Content value="manual" className="space-y-4">
                  <div>
                    <h3 className="text-lg font-medium text-gray-900 mb-2">手動輸入金鑰</h3>
                    <p className="text-sm text-gray-500 mb-6">
                      如果無法掃描 QR Code，您可以手動將下方的金鑰輸入到您的驗證器應用程式中。
                    </p>

                    <div className="space-y-4">
                      <div>
                        <label className="text-sm font-medium text-gray-700 block mb-2">帳戶名稱</label>
                        <div className="flex items-center gap-2">
                          <input
                            className="flex-1 h-10 rounded-md border border-gray-300 bg-gray-50 px-3 py-2 text-sm"
                            value="zhang@example.com"
                            readOnly
                          />
                          <button
                            onClick={() => copyToClipboard("zhang@example.com")}
                            className="p-2 rounded-md hover:bg-gray-100 transition-colors"
                          >
                            <Copy className="h-4 w-4" />
                          </button>
                        </div>
                      </div>

                      <div>
                        <label className="text-sm font-medium text-gray-700 block mb-2">密鑰</label>
                        <div className="flex items-center gap-2">
                          <input
                            className="flex-1 h-10 rounded-md border border-gray-300 bg-gray-50 px-3 py-2 text-sm font-mono"
                            value={secretKey}
                            readOnly
                          />
                          <button
                            onClick={() => copyToClipboard(secretKey)}
                            className={`p-2 rounded-md transition-colors ${
                              copiedSecret ? "bg-green-100 text-green-600" : "hover:bg-gray-100"
                            }`}
                          >
                            {copiedSecret ? <Check className="h-4 w-4" /> : <Copy className="h-4 w-4" />}
                          </button>
                        </div>
                        {copiedSecret && <p className="text-xs text-green-600 mt-1">已複製到剪貼簿</p>}
                      </div>

                      <div className="grid grid-cols-2 gap-4 text-sm">
                        <div>
                          <span className="font-medium text-gray-700">發行者：</span>
                          <span className="text-gray-600">Your App Name</span>
                        </div>
                        <div>
                          <span className="font-medium text-gray-700">類型：</span>
                          <span className="text-gray-600">基於時間 (TOTP)</span>
                        </div>
                      </div>
                    </div>

                    <div className="bg-yellow-50 border border-yellow-200 rounded-lg p-4">
                      <h4 className="text-sm font-medium text-yellow-900 mb-1">重要提醒：</h4>
                      <p className="text-sm text-yellow-800">請確保您的裝置時間設定正確，否則可能導致驗證碼不匹配。</p>
                    </div>
                  </div>
                </Tabs.Content>
              </div>
            </Tabs.Root>
          </div>

          <div className="p-6 border-t border-gray-200 bg-gray-50">
            <button
              onClick={() => setCurrentStep(2)}
              className="inline-flex items-center justify-center rounded-md bg-black px-4 py-2 text-sm font-medium text-white hover:bg-gray-800 focus:outline-none focus:ring-2 focus:ring-black focus:ring-offset-2 transition-colors"
            >
              下一步：驗證設定
            </button>
          </div>
        </div>
      )}

      {/* Step 2: Verification */}
      {currentStep === 2 && (
        <div className="bg-white border border-gray-200 rounded-lg shadow-sm">
          <div className="p-6 border-b border-gray-200">
            <h2 className="text-lg font-medium text-gray-900">驗證設定</h2>
            <p className="text-sm text-gray-500 mt-1">請輸入您的驗證器應用程式顯示的6位數驗證碼。</p>
          </div>

          <div className="p-6 space-y-6">
            <div className="text-center">
              <div className="w-16 h-16 bg-blue-50 rounded-full flex items-center justify-center mx-auto mb-4">
                <Smartphone className="h-8 w-8 text-blue-600" />
              </div>
              <h3 className="text-lg font-medium text-gray-900 mb-2">輸入驗證碼</h3>
              <p className="text-sm text-gray-500 mb-6">
                打開您的驗證器應用程式，找到剛才添加的帳戶，輸入顯示的6位數驗證碼。
              </p>

              <div className="max-w-xs mx-auto">
                <input
                  type="text"
                  maxLength={6}
                  value={verificationCode}
                  onChange={(e) => setVerificationCode(e.target.value.replace(/\D/g, ""))}
                  className="w-full h-12 text-center text-2xl font-mono tracking-widest rounded-md border border-gray-300 bg-white focus:outline-none focus:ring-2 focus:ring-black focus:ring-offset-2 focus:border-transparent transition-all"
                  placeholder="000000"
                />
                <p className="text-xs text-gray-500 mt-2">驗證碼每30秒更新一次</p>
              </div>
            </div>

            <div className="bg-gray-50 border border-gray-200 rounded-lg p-4">
              <h4 className="text-sm font-medium text-gray-900 mb-2">遇到問題？</h4>
              <ul className="text-sm text-gray-600 space-y-1">
                <li>• 確保您的裝置時間設定正確</li>
                <li>• 檢查是否正確掃描了 QR Code 或輸入了金鑰</li>
                <li>• 等待驗證碼更新後再次嘗試</li>
              </ul>
            </div>
          </div>

          <div className="p-6 border-t border-gray-200 bg-gray-50 flex gap-3">
            <button
              onClick={() => setCurrentStep(1)}
              className="inline-flex items-center justify-center rounded-md border border-gray-300 bg-white px-4 py-2 text-sm font-medium text-gray-700 hover:bg-gray-50 focus:outline-none focus:ring-2 focus:ring-black focus:ring-offset-2 transition-colors"
            >
              上一步
            </button>
            <button
              onClick={handleVerification}
              disabled={verificationCode.length !== 6 || isVerifying}
              className="inline-flex items-center justify-center rounded-md bg-black px-4 py-2 text-sm font-medium text-white hover:bg-gray-800 focus:outline-none focus:ring-2 focus:ring-black focus:ring-offset-2 transition-colors disabled:opacity-50 disabled:cursor-not-allowed"
            >
              {isVerifying ? (
                <>
                  <RefreshCw className="h-4 w-4 mr-2 animate-spin" />
                  驗證中...
                </>
              ) : (
                "驗證並完成設定"
              )}
            </button>
          </div>
        </div>
      )}

      {/* Step 3: Backup Codes */}
      {currentStep === 3 && isSetupComplete && (
        <div className="bg-white border border-gray-200 rounded-lg shadow-sm">
          <div className="p-6 border-b border-gray-200">
            <div className="flex items-center gap-2">
              <Check className="h-5 w-5 text-green-600" />
              <h2 className="text-lg font-medium text-gray-900">設定完成！</h2>
            </div>
            <p className="text-sm text-gray-500 mt-1">雙重驗證已成功啟用。請保存以下備用恢復碼。</p>
          </div>

          <div className="p-6 space-y-6">
            <div className="bg-green-50 border border-green-200 rounded-lg p-4">
              <h3 className="text-sm font-medium text-green-900 mb-1">✅ 雙重驗證已啟用</h3>
              <p className="text-sm text-green-800">
                您的帳戶現在受到雙重驗證保護。下次登入時，您需要輸入驗證器應用程式的驗證碼。
              </p>
            </div>

            <div>
              <h3 className="text-lg font-medium text-gray-900 mb-2">備用恢復碼</h3>
              <p className="text-sm text-gray-500 mb-4">
                當您無法使用驗證器應用程式時，可以使用這些備用碼登入。每個備用碼只能使用一次。
              </p>

              <div className="bg-gray-50 border border-gray-200 rounded-lg p-4 mb-4">
                <div className="grid grid-cols-2 gap-2">
                  {backupCodes.map((code, index) => (
                    <code key={index} className="text-sm bg-white px-2 py-1 rounded border font-mono">
                      {code}
                    </code>
                  ))}
                </div>
              </div>

              <div className="flex gap-2">
                <button
                  onClick={downloadBackupCodes}
                  className="inline-flex items-center justify-center rounded-md border border-gray-300 bg-white px-3 py-2 text-sm font-medium text-gray-700 hover:bg-gray-50 focus:outline-none focus:ring-2 focus:ring-black focus:ring-offset-2 transition-colors"
                >
                  <Download className="h-4 w-4 mr-2" />
                  下載備用碼
                </button>
                <button
                  onClick={() => copyToClipboard(backupCodes.join("\n"))}
                  className="inline-flex items-center justify-center rounded-md border border-gray-300 bg-white px-3 py-2 text-sm font-medium text-gray-700 hover:bg-gray-50 focus:outline-none focus:ring-2 focus:ring-black focus:ring-offset-2 transition-colors"
                >
                  <Copy className="h-4 w-4 mr-2" />
                  複製備用碼
                </button>
              </div>
            </div>

            <div className="bg-red-50 border border-red-200 rounded-lg p-4">
              <h4 className="text-sm font-medium text-red-900 mb-1">⚠️ 重要提醒</h4>
              <ul className="text-sm text-red-800 space-y-1">
                <li>• 請將備用碼保存在安全的地方</li>
                <li>• 不要將備用碼存放在與此帳戶相同的裝置上</li>
                <li>• 每個備用碼只能使用一次</li>
                <li>• 建議定期更新備用碼</li>
              </ul>
            </div>
          </div>

          <div className="p-6 border-t border-gray-200 bg-gray-50">
            <button
              onClick={onComplete}
              className="inline-flex items-center justify-center rounded-md bg-black px-4 py-2 text-sm font-medium text-white hover:bg-gray-800 focus:outline-none focus:ring-2 focus:ring-black focus:ring-offset-2 transition-colors"
            >
              完成設定
            </button>
          </div>
        </div>
      )}
    </div>
  )
}
